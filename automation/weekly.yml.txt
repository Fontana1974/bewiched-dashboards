name: Weekly dashboard refresh
# Fully automated weekly run — no agent, no desktop.
# Enable the schedule below once GCP_SA_JSON secret is set and run_weekly.py is finished+tested.
on:
  workflow_dispatch: {}          # lets you run it on-demand from the Actions tab
  # schedule:
  #   - cron: '0 9 * * 1'        # 09:00 UTC Mon = 10:00 UK (BST). Use '30 9 * * 1' for 10:30 UK.
permissions:
  contents: write                # built-in GITHUB_TOKEN can commit/push — no PAT needed
concurrency:
  group: weekly-refresh
  cancel-in-progress: false
jobs:
  refresh:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-python@v5
        with: { python-version: '3.12' }
      - run: pip install -r requirements.txt
      - name: Build dashboards
        env:
          GCP_SA_JSON: ${{ secrets.GCP_SA_JSON }}
        run: python run_weekly.py
      - name: Commit & push
        run: |
          git config user.name  "Bewiched Dashboards Bot"
          git config user.email "matt@bewiched.co.uk"
          git add -A
          git commit -m "Weekly automated refresh $(date +%F)" || { echo "no changes"; exit 0; }
          git push
